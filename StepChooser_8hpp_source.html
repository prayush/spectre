32 /// \ingroup TimeGroup
33 ///
34 /// \brief StepChoosers suggest upper bounds on step sizes.
35 ///
36 /// Concrete StepChoosers should define `operator()` returning the
37 /// magnitude of the desired step (as a double).
38 ///
39 /// Derived classes must indicate whether the chooser is usable as a
40 /// step chooser, slab chooser, or both by inheriting from StepChooser
41 /// with the appropriate `StepChooserUse` template argument. A class
42 /// cannot inherit from both base classes simultaneously; if both uses
43 /// are supported the use must be chosen using a template parameter.
44 /// If a derived class is usable as a step chooser (i.e., inherits
45 /// from StepChooser<StepChooserUse::LtsStep>), it must specify type
46 /// aliases `argument_tags` and `return_tags` for the arguments to the
47 /// call operator. If it is usable only for slab choosing (i.e.,
48 /// unconditionally inherits from StepChooser<StepChooserUse::Slab>),
49 /// it need only specify `argument_tags`, as slab choosers are not
50 /// permitted mutable access to the \ref DataBoxGroup "DataBox".
51 ///
52 /// For the class interface, see the specializations
53 /// StepChooser<StepChooserUse::LtsStep> and
54 /// StepChooser<StepChooserUse::Slab>.
55 template <typename StepChooserUse>
56 class StepChooser;

109  /// The `last_step_magnitude` parameter describes the slab size to be
110  /// adjusted; It may be infinite if the appropriate size cannot be determined.
111  ///
112  /// As opposed to the decision made by `desired_step` in the
113  /// StepChooser<StepChooserUse::LtsStep> specialization, the slab
114  /// change decision must be callable from an event (so cannot store
115  /// state information in the \ref DataBoxGroup "DataBox"), and we do
116  /// not have the capability to reject a slab so this function
117  /// returns only a `double` indicating the desired slab size.
118  template <typename Metavariables, typename DbTags>
119  double desired_slab(
120  const double last_step_magnitude,
121  const db::DataBox<DbTags>& box,
122  const Parallel::GlobalCache<Metavariables>& cache) const noexcept {
123  ASSERT(last_step_magnitude > 0.,
124  "Passed non-positive step magnitude: " << last_step_magnitude);
125  using factory_classes =
126  typename std::decay_t<decltype(db::get<Parallel::Tags::Metavariables>(
127  box))>::factory_creation::factory_classes;
128  const auto result =
129  call_with_dynamic_type<std::pair<double, bool>,
130  tmpl::at<factory_classes, StepChooser>>(
131  this, [&last_step_magnitude, &box,
132  &cache](const auto* const chooser) noexcept {
133  return db::apply(*chooser, box, last_step_magnitude, cache);
134  });
135  ASSERT(
136  result.first > 0.,
137  "StepChoosers should always return positive values. Got " << result);
138  return result.first;
139  }
140 };

158  /// The `last_step_magnitude` parameter describes the step size to be
159  /// adjusted. It may be the step size or the slab size, or may be
160  /// infinite if the appropriate size cannot be determined.
161  ///
162  /// The return value of this function contains the desired step size
163  /// and a `bool` indicating whether the step should be accepted. If the `bool`
164  /// is `false`, the current time step will be recomputed with a step size
165  /// informed by the desired step value returned by this function. The
166  /// implementations of the call operator in derived classes should always
167  /// return a strictly smaller step than the `last_step_magnitude` when they
168  /// return `false` for the second member of the pair (indicating step
169  /// rejection).
170  template <typename Metavariables, typename DbTags>
171  std::pair<double, bool> desired_step(
172  const gsl::not_null<db::DataBox<DbTags>*> box,
173  const double last_step_magnitude,
174  const Parallel::GlobalCache<Metavariables>& cache) const noexcept {
175  ASSERT(last_step_magnitude > 0.,
176  "Passed non-positive step magnitude: " << last_step_magnitude);
177  using factory_classes =
178  typename std::decay_t<decltype(db::get<Parallel::Tags::Metavariables>(
179  *box))>::factory_creation::factory_classes;
180  const auto result =
181  call_with_dynamic_type<std::pair<double, bool>,
182  tmpl::at<factory_classes, StepChooser>>(
183  this, [&last_step_magnitude, &box,
184  &cache](const auto* const chooser) noexcept {
185  using chooser_type = typename std::decay_t<decltype(*chooser)>;
186  return db::mutate_apply<typename chooser_type::return_tags,
187  typename chooser_type::argument_tags>(
188  *chooser, box, last_step_magnitude, cache);
189  });
190  ASSERT(
191  result.first > 0.,
192  "StepChoosers should always return positive values. Got " << result);
193  return result;
194  }
195 };
